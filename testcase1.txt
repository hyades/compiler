%Following function reads marks of a student in all subjects and returns as a record variable

% atleast one input parameter

output parameter list [ record #marks b3c45, int b3b4];
	type record #marks:d3 : global;
	type int : b3b4;

	b3c45 <--- c3b5 + c3b6;

end
